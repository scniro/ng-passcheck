module.config(['passCheckProvider', function (passCheckProvider) {
	passCheckProvider.init({
		policies: { // optional. defaults shown. shortcut syntax such as \d must be escaped as \\d
			medium: {
				pattern: '^(?=.*?[A-Z])(?=.*?[a-z])(?=.*?[0-9])',
				minimum: 8
			},
			strong: {
				pattern: '^(?=.*?[A-Z])(?=.*?[a-z])(?=.*?[0-9])(?=.*?[^\\w\\s])',
				minimum: 12
			}
		},
		testCommon: { // do not specify this configuration to skip common checks
			path: '~/passwords.json', // optionally, passwords-hashed.json
			limit: 10000
		},
		increments: { // optional. defaults shown.
			base: 1,
			bonus: {
				medium: 1.25
				strong: 1.50
			}
		}
	});
}]);
